Exercício 16: Faça um programa para calcular a área de um triângulo e que não permita a entrada de dados inválidos, ou seja,
 medidas menores ou iguais a zero.
#include <stdio.h>

#include <stdlib.h>


int main(){
	
int base, altura, area;
	do
	
{
		
printf("\nEntre com a base do triangulo: ");
	
	scanf("%d", &base);
	
	printf("\nEntre com a altura do triangulo: ");
		
        scanf("%d", &altura);
		
        area= (base*altura) / 2;
