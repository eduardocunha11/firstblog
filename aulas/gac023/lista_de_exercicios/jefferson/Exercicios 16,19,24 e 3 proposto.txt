#include <stdio.h>
#include <stdlib.h>

int main(){
	int base, altura, area;
	do
	{
		printf("\nEntre com a base do triangulo: ");
		scanf("%d", &base);
		printf("\nEntre com a altura do triangulo: ");
		scanf("%d", &altura);
		area= (base*altura) / 2;
		printf("\nArea eh: %d", area);
	} while (base > 0 && altura > 0);
	return 0;
}
