/* 1- Um funcionário de uma empresa recebe, anualmente, aumento salaral. sabe-se que:
a) esse funcionário foi contratado em 2005, com salário de R$1000,00.
b) em 2006,ele recebeu aumento de 1,5% sobre seu saalário inicial.
c)A partir de 2007, os aumentos salariais sempre corresponderam ao dobro do percentual do ano anterior.
*/
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int i, ano_atual;
    int ano_entrada= 2006;
    float  salario=100;
    float percentual=0.015;
    printf ("Digite o ano atual: \n");
    scanf ("%i", &ano_atual);

    for (i= ano_entrada; i <= ano_atual; i++){
        salario += percentual * salario;
        percentual = percentual *2;
    }
    return 0;
}

/*3- Faça um programa que leia o numero Nque indica quantos valores inteiros e positivos devem ser lidos a seguir.
para cada número lido,mostre uma tabela contendo o valor lido e o fatorial desse valor.
*/
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int fat=1;
    int i,num;
    printf ("Digite um numero: \n");
    scanf ("%i", &num);
    for (i=num; num>1; num--)
        fat*= num;
    printf ("Fatorial=  %i\n", fat);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main() {

  int i, x;
  int qtd = 0;

  do {

    printf("Digite um número inteiro e positivo: \n");
    scanf("%d", &x);
  } while (x <= 0);

  for (i = 1; i <= x; i++) {
    if (x % i == 0) {
     qtd++;
    }
  }

  if (qtd == 2)
    printf("Primo!\n" );
  else
    printf("Nao primo!\n");

  return 0;
}
