/* 7- Faça um programa que decide dar um aumento de 30% nos salarios inferiores a R$500.
Faca um programar que receba o salário do funcionário e mostre o valor do salario
reajustado ou uma mensagem, caso ele não tenha direito de aumento.
 */

#include <stdio.h>
#include <stdlib.h>

int main()
{
 float sal, salreaj;
 printf ("Digite seu salario:\n");
 scanf ("%f", &sal );
 if (sal<= 500){
    salreaj = sal + (sal * 0.30);
    printf ("Novo salario = %f\n", salreaj);
 }
 else {
    printf ("Nao possui aumento.");
 }

    return 0;
}
