/* 16- Faça um programa para calcular a área de 
um trinângulo sem a entrada de dados invalidos.*/

#include <stdio.h>
#include <stdlib.h>
int main ()
{
float base=0, altura=0, area=0;

do{
    printf("Digite  a base: \n");
    scanf("%f", &base);
    printf ("Digite a altura: ");
    scanf("%f", &altura);

    if (base <=0 || altura <=0){
        printf("\nNumero invalido!\n");
    }

} while (base <=0 || altura <=0);

area = (base*altura)/2;
printf ("Area do triangulo = %f", area);

return 0;
}
