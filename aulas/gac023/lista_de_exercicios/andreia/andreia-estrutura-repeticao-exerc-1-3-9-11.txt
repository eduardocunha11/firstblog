#include<stdio.h>
#include<stdlib.h>
#include<math.h>
int main ()
{
    int i,anoatual, salario;
    float novosal,percentual;
    printf("Digite o ano atual:");
    scanf("%i", &anoatual);
    salario = 1000;
    percentual=(1.5/100);
    novosal= salario+percentual*salario;
    for (i=2007; i<=anoatual; i++){

        novosal=novosal+percentual*novosal;

}

#include<stdio.h>
#include<stdlib.h>
#include<math.h>
int main ()
{
    int num,fat,i,j,n;
    printf("Digite o valor de n:\n");
    scanf("%i", &n);
    for (i=0 ; i<n; i++){
        printf("\nDigite um numero para ser calculado o fatorial:\n");
        scanf("%i", &num);
        fat=1;
        for (j=1; j<=num; j++){
            fat= fat*j;
        }
        printf("O resultado do fatorial e: %i", fat);
    }
return 0;
}
