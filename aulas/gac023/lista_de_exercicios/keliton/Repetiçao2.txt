            printf("Opcao Invalida");

            if(sala<500)
                imp=(sala*0.05);

                    imp=(sala*0.10);
                    else if(sala>850)
                        imp=(sala*0.15);

        }
        if(op==2){
            printf("\nDigite o salario:");
            scanf("%f",&sala);

            if(sala>1500)
                novo=sala+25;

                novo=sala+50;

         if(op==3){
