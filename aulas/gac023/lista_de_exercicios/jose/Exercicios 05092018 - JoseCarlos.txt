#include <stdio.h>
#include <stdlib.h>
int main(){
	int Base, Altura, Area_Total;
	do
	{
	printf("\n Digite o valor da base do triangulo: ");
	scanf("%d", &Base);
	printf("\n Digite o valor da altura do triangulo: ");
	scanf("%d", &Altura);
	Area_Total= (Base * Altura) / 2;
	printf("\n Area Total: %d", Area_Total);
	} while (Base > 0 && Altura > 0);
	return 0;
}
