#include <stdio.h>
#include <stdlib.h>

int main(){
	float aumento = 0, salario;
	salario = 1000;
	int ano, i;
	printf("Digite o ano: ");
	scanf("%d", &ano);
	if(ano > 2006){
		aumento = 0.015;
	}
