{

#include <stdio.h>
#include <stdlib.h>

int main()
{
int vet[10], i, j, aux;

// Enquanto i for igual a 1 e menor ou igual a 10, peca ao usuario para digitar valores e armazenar no vetor i.

for (i=0; i<10; i++)
{
    printf("Digite um valor:");
    scanf("%d", &vet[i]);
}

// Para i obedecendo essa condição faca um segundo vetor que irá armazenar a ordem correta do numeros e que esteja dentro da condicao do vetor i
for (i=0; i<10; i++)
{
    for(j=0; j<=10;j++)
    {
        if (vet[j] < vet[j+1]) // se o valor na posicao j for menor que j+1
        {
            aux=vet[j]; //adicione o valor na variavel aux
            vet[j]=vet[j+1]; // adicione o valor a frente no vet j
            vet[j+1]=aux; // o valor na posicao da frente sera armazenado sobre a variavel aux
        }
    }
}
// Feito o vetor j com as posições corretas print o vetor i com os numeros digitados na ordem crescente
for (i=0; i<10; i++)
{
    printf(" %d \t", vet[i]);

  return 0;
}
