#include <stdio.h>
#include <stdlib.h>

int main() {
  float base = 0, altura = 0, area = 0;

  do {
     printf("Informe a base do triângulo: ");
     scanf("%f", &base);
     printf("Informe a altura do triângulo: ");
     scanf("%f", &altura);

     if (base <= 0 || altura <= 0) {
         printf("\nENTRADA INVÁLIDA!\n\n");
     }
  } while(base <= 0 || altura <= 0);

  area = (base * altura)/2;

  printf("\nÁrea do triângulo: %.2f", area);

  return 0;
}
